<script>
async function go() {
  let p = document.getElementById("prompt").value;

  document.getElementById("msg").innerText = "Generating... please wait";

  let res = await fetch("/generate", {
    method: "POST",
    headers: {
      "Content-Type": "application/json"
    },
    body: JSON.stringify({ prompt: p })
  });

  let data = await res.json();

  if (data.video_url) {
    document.getElementById("v").src = data.video_url;
    document.getElementById("msg").innerText = "Done!";
  } else {
    document.getElementById("msg").innerText = "Error: " + data.message;
  }
}
</script>
